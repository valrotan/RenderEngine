set(CMAKE_VERBOSE_MAKEFILE ON)

cmake_minimum_required( VERSION 3.5 )
list(APPEND CMAKE_MODULE_PATH "${CMAKE_CURRENT_LIST_DIR}/cmake")

project( renderEngine LANGUAGES C )
add_executable( renderEngine gtk.c )

find_package(PkgConfig REQUIRED)

pkg_check_modules(GTK3 REQUIRED gtk+-3.0)
message( STATUS "GTK3_FOUND=${GTK3_FOUND}" )
include_directories(${GTK3_INCLUDE_DIRS})
target_link_directories(renderEngine  PRIVATE ${GTK3_LIBRARY_DIRS})
target_link_libraries(renderEngine ${GTK3_LIBRARIES})
